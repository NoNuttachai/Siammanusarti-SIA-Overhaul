sia_update_corruption_seffect = {
	add_to_variable = { SIA_CORRUPTION_TRACKER_VAR = SIA_CSI_LOWER_VAR_TEMP tooltip = SIA_CORRUPTION_TRACKER_VAR__UPDATE.TT }
	clamp_variable = {
		var = SIA_CORRUPTION_TRACKER_VAR
		min = 0
		max = 100
	}
}

SIA_naresuan_fleet_creation = {
	IF = {
		limit = { 
			has_dlc = "Man the Guns"
		}
		create_equipment_variant = {	
			name = "Naresuan Class" 		
			type = ship_hull_light_1
			#name_group = ITA_DD_HISTORICAL
			parent_version = 0
			modules = {
				fixed_ship_battery_slot = ship_light_battery_1
				fixed_ship_anti_air_slot = ship_anti_air_1
				fixed_ship_fire_control_system_slot = ship_fire_control_system_0
				fixed_ship_radar_slot = empty
				fixed_ship_engine_slot = light_ship_engine_1
				fixed_ship_torpedo_slot = ship_torpedo_1
				mid_1_custom_slot = empty
				rear_1_custom_slot = empty
			}
			obsolete = yes
		}
		
		create_ship = {
			type = ship_hull_light_1
			equipment_variant = "Naresuan Class"
			creator = ITA
			name = "HTMS Taskin"
		}		
		create_ship = {
			type = ship_hull_light_1
			equipment_variant = "Naresuan Class"
			creator = ITA
			name = "HTMS Naresuan"
		}	
	}
	ELSE = {
		create_equipment_variant = {
			name = "Naresuan Class"
			type = destroyer_1
			upgrades = {
				ship_torpedo_upgrade = 1
				destroyer_engine_upgrade = 1
				ship_ASW_upgrade = 1
				ship_anti_air_upgrade = 1
			}
			obsolete = yes
		}
		create_ship = {
			type = destroyer_1
			equipment_variant = "Naresuan Class"
			creator = "ITA"
			name = "HTMS Taskin"
		}
		create_ship = {
			type = destroyer_1
			equipment_variant = "Naresuan Class"
			creator = "ITA"
			name = "HTMS Naresuan"
		}
	}
}
